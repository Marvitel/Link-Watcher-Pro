=== Schema groups ===
CREATE TABLE groups (
        id integer not null primary key autoincrement,
        uuid text not null unique,
        agent_id integer,
        kind text not null,
        name text not null,
        description text,
        should_run_items boolean not null,
        is_proxy boolean not null,
        configsets text not null,
        config json not null,
        config_priority integer not null,
        created_by_network_rule_id integer,
        data json,
        created_at integer not null default 0,
        updated_at integer not null default 0,
        foreign key (agent_id) references agents(id),
        foreign key (created_by_network_rule_id) references network_discovery_rules(id)
);
CREATE INDEX group_created_by_network_rule_idx on groups(created_by_network_rule_id);
CREATE INDEX group_agent_idx on groups(agent_id);
CREATE INDEX group_kind_idx on groups(kind);
=== group_parents (hierarquia) ===
CREATE TABLE group_parents (
        id integer not null primary key,
        group_id integer not null,
        parent_id integer not null,
        foreign key (group_id) references groups(id),
        foreign key (parent_id) references groups(id)
);
CREATE INDEX group_parents_group on group_parents(group_id);
CREATE INDEX group_parents_parent on group_parents(parent_id);
