CREATE TABLE events (
        id integer not null primary key,
        correlation_id varchar not null,
        execution_id varchar not null,
        source_event_id integer,
        close_event_id integer,
        container_id integer not null,
        user_id varchar,
        time datetime not null,
        event_type varchar not null,
        closed boolean not null,
        duration_secs integer,
        props varchar not null,
        level integer not null,
        acknowledged_by varchar,
        action_count integer not null,
        comment_count integer not null,
        send_alerts boolean not null default 0
    );
CREATE INDEX idx_events_container_id on events(container_id);
CREATE INDEX idx_events_correlation_id on events(correlation_id);
CREATE INDEX idx_events_execution_id on events(execution_id);
CREATE INDEX idx_events_closed on events(closed);
CREATE INDEX idx_events_event_type on events(event_type);
CREATE INDEX idx_events_time on events(time);
CREATE INDEX idx_events_event_type_time on events(event_type, time);
CREATE INDEX idx_events_container_id_event_type_time on events(container_id, event_type, time);
CREATE TABLE actions (
        id integer not null primary key,
        event_id integer not null references events(id),
        container_id integer,
        action_uuid varchar not null,
        time datetime not null,
        status number not null,
        props varchar not null,
        script varchar,
        result varchar,
        error varchar,
        output varchar
    );
CREATE INDEX idx_actions_event_id on actions(event_id);
CREATE INDEX idx_actions_container_id on actions(container_id);
CREATE TABLE comments (
        id integer not null primary key,
        event_id integer not null references events(id),
        user varchar not null,
        time datetime not null,
        comment varchar not null
    );
CREATE INDEX idx_comments_event_id on comments(event_id);
